		<sf-radio name="obj" :label="obj2.name" :value="obj2" v-model="value"/>
		<sf-radio name="obj" :label="obj3.name" :value="obj3" v-model="value"/>
		<sf-radio name="obj" :label="obj4.name" :value="obj4" v-model="value"/>

	</template>

export default {

			},

				id: 4,

			},
			value: null
		}

</script>
